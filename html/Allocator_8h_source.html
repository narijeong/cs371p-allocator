198  pointer allocate(size_type n) {
199  if((int)n <= 0)
200  throw std::bad_alloc();
201 
202  n = n * sizeof(T);
203  SENTINEL_TYPE freespace = (*this)[0];
204  int position = 0;
205 
206  /* find first fit */
207  while(position < N && (int)freespace < (int)n) {
208  position += (abs_val(freespace) + 2 * sizeof(SENTINEL_TYPE));
209  freespace = (*this)[position];
210  }
211 
212  if (position >= N)
213  throw std::bad_alloc();
214 
215  int total = (*this)[position] + 2 * sizeof(SENTINEL_TYPE);
216  int required = total - (int)n - 2 * sizeof(SENTINEL_TYPE);
217  int second_sentinel_value = required - 2 * sizeof(SENTINEL_TYPE);
218 
219  /* allocating whole block */
220  if(second_sentinel_value < (int)sizeof(T)) {
221  int sentinel = (*this)[position];
222  (*this)[position] = sentinel * -1;
223  (*this)[position + sizeof(SENTINEL_TYPE) + sentinel] = sentinel * -1;
224  }
225  else {
226  /* spliting the block into two blocks */
227  (*this)[position] = -1 * n;
228  (*this)[position + sizeof(SENTINEL_TYPE) + n] = -1 * n;
229  (*this)[position + 2 * sizeof(SENTINEL_TYPE) + n] = second_sentinel_value;
230  (*this)[position + 3 * sizeof(SENTINEL_TYPE) + n + second_sentinel_value] = second_sentinel_value;
231  }
232 
233  assert(valid());
234  return (pointer) &(*this)[position + sizeof(SENTINEL_TYPE)];
235  }
